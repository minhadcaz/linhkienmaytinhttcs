import React, { createContext, useState, useContext, ReactNode, useEffect } from "react";
import { AuthContext } from "./authcontext";
import axios from "axios";
import message from "antd/es/message";


// Định nghĩa kiểu dữ liệu cho Sản phẩm trong giỏ
export interface CartItem {
  id: number;
  name: string;
  category: string;
  price: number;
  originalPrice?: number | null;
  quantity: number;
}

// Định nghĩa các hàm có trong Context
interface CartContextType {
  cartItems: any[];
  totalItemCount: number;
  fetchCart: () => void;
  addToCart: (productId: string, quantity?: number) => void;
}

const CartContext = createContext<CartContextType | undefined>(undefined);

export const CartProvider: React.FC<{ children: ReactNode }> = ({
  children,
}) => {
  const [cartItems, setCartItems] = useState<CartItem[]>([]);
  const [totalItemCount, setTotalItemCount] = useState<number>(0);
  const { isAuthenticated } = useContext(AuthContext); // Kiểm tra xem user đăng nhập chưa
  const fetchCart = async () => {
    if (!isAuthenticated) return;
    try {
      const token = localStorage.getItem('access_token');
      const response = await axios.get('http://localhost:8080/api/cart', {
        headers: { Authorization: `Bearer ${token}` }
      });
      setCartItems(response.data.chitietgiohang || []);
      setTotalItemCount(response.data.sum || 0); // Lấy thuộc tính sum từ bảng giohang
    } catch (error) {
      console.error("Lỗi lấy giỏ hàng", error);
    }
  };

  // Tự động fetch giỏ hàng ngay khi User đăng nhập thành công
  useEffect(() => {
    fetchCart();
  }, [isAuthenticated]);

  // Hàm thêm vào giỏ hàng
  const addToCart = async (productId: string, quantity: number = 1) => {
    if (!isAuthenticated) {
      message.warning("Vui lòng đăng nhập để mua hàng!");
      return;
    }
    try {
      const token = localStorage.getItem('access_token');
      await axios.post('http://localhost:8080/api/cart/add',
        { productId, quantity },
        { headers: { Authorization: `Bearer ${token}` } }
      );
      message.success("Đã thêm sản phẩm vào giỏ hàng!");
      fetchCart(); // Gọi lại API để cập nhật Navbar lập tức
    } catch (error) {
      message.error("Lỗi khi thêm vào giỏ hàng");
    }
  };


  // // Hàm Thêm vào giỏ (Mô phỏng logic Upsert của Backend)
  // const addToCart = (product: Omit<CartItem, "quantity">) => {
  //   setCartItems((prev) => {
  //     const existingItem = prev.find((item) => item.id === product.id);
  //     if (existingItem) {
  //       // Nếu đã có, tăng số lượng
  //       return prev.map((item) =>
  //         item.id === product.id
  //           ? { ...item, quantity: item.quantity + 1 }
  //           : item,
  //       );
  //     }
  //     // Nếu chưa có, thêm mới với quantity = 1
  //     return [...prev, { ...product, quantity: 1 }];
  //   });
  // };

  const removeFromCart = (id: number) => {
    setCartItems((prev) => prev.filter((item) => item.id !== id));
  };

  const updateQuantity = (id: number, delta: number) => {
    setCartItems((prev) =>
      prev.map((item) => {
        if (item.id === id) {
          const newQuantity = item.quantity + delta;
          return { ...item, quantity: newQuantity > 0 ? newQuantity : 1 };
        }
        return item;
      }),
    );
  };

  return (
    <CartContext.Provider
      value={{ cartItems, totalItemCount, fetchCart, addToCart }}
    >
      {children}
    </CartContext.Provider>
  );
};

// Hook tùy chỉnh để sử dụng dễ dàng hơn
export const useCart = () => {
  const context = useContext(CartContext);
  if (!context) throw new Error("useCart phải được bọc bên trong CartProvider");
  return context;
};
